# Log likelihood function 

log_lik = function(b_numlang){
  
  # linear predictor
  eta = b0 +
         b_numlang * newfearon$numlang +
         b_polity * newfearon$polity2 +
         b_logpop * newfearon$log_pop
  
  # predicted probabilities
  p = 1 / (1 + exp(-eta))
  
  # log-likelihood
  loglike = sum(newfearon$any_onset * log(p) + 
            (1 - newfearon$any_onset) * log(1 - p))
  
  return(loglike)

The MLE is the value of \(\beta_{\text{numlang}}\) that makes the observed pattern of 0s and 1s most probable. The log-likelihood peaks at the MLE because this is the value of the coefficient that makes the observed outcomes most consistent with the model’s predicted probabilities. Moving away from this value reduces the likelihood because the model’s predictions become less aligned with the observed data.

3.b

When the slope is positive, the algorithm would step to the right; when negative, it would step to the left. The slopes to the left of the MLE are positive and those to the right are negative, correctly pointing toward the MLE.

3.c

The slope is largest far from the MLE and smallest near it. This is because the log-likelihood curve is steep far away from the peak and flat near the maximum. An algorithm should will large steps when the slope is large and small steps when the slope is small.

Understanding the Hessian: how fast the curve is moving.

# Expand the previous function to include Hessian

epsilon = 0.0001

compute_all = function(b0) {
  
  log_0 = log_lik(b0)
  log_1 = log_lik(b0 + epsilon)
  log_2 = log_lik(b0 + 2 * epsilon)
  
  slope1 = (log_1 - log_0) / epsilon
  slope2 = (log_2 - log_1) / epsilon
  
  hessian = (slope2 - slope1) / epsilon
  
  return(data.frame(
    beta = b0,
    log_beta = log_0,
    slope = slope1,
    hessian = hessian
  ))
}

loglike = function(onset_mod, newfearon) {
  
  coefs = coef(onset_mod)
  
  b0 = coefs["(Intercept)"]
  b_polity = coefs["polity2"]
  b_logpop = coefs["log_pop"]
  
  function(b_numlang) {
    
    eta = b0 +
      b_numlang * newfearon$numlang +
      b_polity * newfearon$polity2 +
      b_logpop * newfearon$log_pop
    
    p = 1 / (1 + exp(-eta))
    
    sum(newfearon$any_onset * log(p) +
          (1 - newfearon$any_onset) * log(1 - p))
  }
}
